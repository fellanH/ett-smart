   import re

   def detect_input_type(query: str) -> str:
       """Auto-detect if input is org number or company name."""
       org_pattern = r'^\d{6}-?\d{4}$'
       cleaned = query.strip().replace(' ', '')
       if re.match(org_pattern, cleaned):

       return "company_name"
